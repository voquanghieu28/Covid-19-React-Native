// Approach 1 // making push operation costly class Stack { constructor() { this.q1 = new Queue(); this.q2 = new Queue(); this.size = 0; } push(val) { let num = this.q1.size; this.q2.enqueue(val); for(let i = 0; i < num; i++){ let nextVal = this.q1.dequeue(); this.q2.enqueue(nextVal); } let temp = this.q1; this.q1 = this.q2; this.q2 = temp; this.size++; return this; } pop() { this.size--; return this.q1.dequeue(); } } // Approach 2 // making pop operation costly class Stack { constructor() { this.q1 = new Queue(); this.q2 = new Queue(); this.size = 0; } push(val) { this.q1.enqueue(val); this.size++; return this; } pop() { let num = this.q1.size; for(let i = 0; i < num - 1; i++){ let nextVal = this.q1.dequeue(); this.q2.enqueue(nextVal); } let temp = this.q1; this.q1 = this.q2; this.q2 = temp; this.size--; return this.q2.dequeue(); } }

class Stack { constructor() { this.queue = new Queue(); } push(val) { let queue = this.queue ,queueTwo = new Queue(); queueTwo.enqueue(val); while(queue.size){ queueTwo.enqueue(queue.dequeue()); } this.queue = queueTwo; return this; } //O(n) time complexity pop() { return this.queue.dequeue(); } //O(1) time complexity } // QUEUE AND NODE HAVE BEEN IMPLEMENTED FOR YOU class Node { constructor(value) { this.value = value; this.next = null; } } class Queue { constructor() { this.first = null; this.last = null; this.size = 0; } enqueue(data) { var node = new Node(data); if (!this.first) { this.first = node; this.last = node; } else { this.last.next = node; this.last = node; } return ++this.size; } dequeue() { if (!this.first) return null; var temp = this.first; if (this.first == this.last) { this.last = null; } this.first = this.first.next; this.size--; return temp.value; } }

Solution to stack with 2 Queues
class Stack {    constructor() {        this.first = null;        this.last = null;        this.size = 0;        this.queueOne = new Queue();        this.queueTwo = new Queue();    }    push(val) {        this.queueOne.enqueue(val);        this.first = this.queueOne.first;        this.last = this.queueTwo.last;        this.size++;        return this;    }    pop() {        var poppedNode = null;        if (this.queueOne.size > 0) {            if (this.queueOne.size === 1) {                poppedNode = this.queueOne.dequeue();                this.first = null;                this.last = null;            }            else {                var currentNode = this.queueOne.dequeue();                for (let i=0; i<=this.queueOne.size;i++) {                    this.queueTwo.enqueue(currentNode);                    currentNode = this.queueOne.dequeue();                }                poppedNode = currentNode;                while (this.queueTwo.size > 0) {                    let poppedNodeFromQueueTwo = this.queueTwo.dequeue();                    this.queueOne.enqueue(poppedNodeFromQueueTwo);                }            }            this.size--;        }        return poppedNode;     }}// QUEUE AND NODE HAVE BEEN IMPLEMENTED FOR YOUclass Node {    constructor(value) {        this.value = value;        this.next = null;    }}class Queue {    constructor() {        this.first = null;        this.last = null;        this.size = 0;    }    enqueue(data) {        var node = new Node(data);        if (!this.first) {            this.first = node;            this.last = node;        } else {            this.last.next = node;            this.last = node;        }        return ++this.size;    }    dequeue() {        if (!this.first) return null;        var temp = this.first;        if (this.first == this.last) {            this.last = null;        }        this.first = this.first.next;        this.size--;        return temp.value;    }}
